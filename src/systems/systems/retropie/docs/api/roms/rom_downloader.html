NES ROM 下载和传输工具

这是一个自动化ROM下载和传输工具，专门用于RetroPie游戏系统。 支持从合法ROM资源站下载游戏合集并自动传输到树莓派。

主要功能：

  • 从Archive.org等合法资源站搜索和下载ROM
  • 支持断点续传和校验和验证
  • 自动解压ZIP格式的ROM文件
  • 通过SFTP自动传输到树莓派
  • 完整的日志记录和错误处理
  • 配置文件驱动的灵活配置

系统要求：

  • Python 3.7+
  • 网络连接
  • 树莓派SSH访问权限

212    def download_file(self, url: str, filename: str) -> Optional[Path]:
213        """下载文件，支持断点续传"""
214        file_path = self.download_dir / filename
215        
216        # 检查是否已存在
217        if file_path.exists():
218            logger.info(f"文件已存在: {file_path}")
219            return file_path
220        
221        try:
222            # 获取文件大小
223            response = self.session.head(url, timeout=30)
224            response.raise_for_status()
225            
226            total_size = int(response.headers.get('content-length', 0))
227            
228            # 检查是否有部分下载的文件
229            temp_path = file_path.with_suffix('.tmp')
230            resume_pos = 0
231            
232            if temp_path.exists():
233                resume_pos = temp_path.stat().st_size
234                if resume_pos >= total_size:
235                    # 文件已完整下载
236                    temp_path.rename(file_path)
237                    logger.info(f"文件下载完成: {file_path}")
238                    return file_path
239            
240            # 设置断点续传头
241            headers = {}
242            if resume_pos > 0:
243                headers['Range'] = f'bytes={resume_pos}-'
244                logger.info(f"断点续传: {resume_pos} bytes")
245            
246            # 开始下载
247            response = self.session.get(url, headers=headers, stream=True, timeout=30)
248            response.raise_for_status()
249            
250            mode = 'ab' if resume_pos > 0 else 'wb'
251            
252            with open(temp_path, mode) as f:
253                with tqdm(
254                    total=total_size,
255                    initial=resume_pos,
256                    unit='B',
257                    unit_scale=True,
258                    desc=filename
259                ) as pbar:
260                    
261                    for chunk in response.iter_content(chunk_size=self.config["download"]["chunk_size"]):
262                        if chunk:
263                            f.write(chunk)
264                            pbar.update(len(chunk))
265            
266            # 重命名临时文件
267            temp_path.rename(file_path)
268            logger.info(f"下载完成: {file_path}")
269            return file_path
270            
271        except Exception as e:
272            logger.error(f"下载失败: {e}")
273            # 清理临时文件
274            if temp_path.exists():
275                temp_path.unlink()
276            return None

376    def upload_roms(self, rom_files: List[Path]) -> bool:
377        """上传ROM文件到树莓派"""
378        if not rom_files:
379            logger.warning("没有ROM文件需要上传")
380            return False
381        
382        ssh = self.connect_sftp()
383        if not ssh:
384            return False
385        
386        try:
387            sftp = ssh.open_sftp()
388            pi_config = self.config["raspberry_pi"]
389            remote_path = pi_config["roms_path"]
390            
391            # 确保远程目录存在
392            try:
393                sftp.stat(remote_path)
394            except FileNotFoundError:
395                logger.info(f"创建远程目录: {remote_path}")
396                sftp.mkdir(remote_path)
397            
398            uploaded_count = 0
399            
400            for rom_file in rom_files:
401                try:
402                    remote_file = f"{remote_path}/{rom_file.name}"
403                    
404                    # 检查文件是否已存在
405                    try:
406                        remote_stat = sftp.stat(remote_file)
407                        local_stat = rom_file.stat()
408                        
409                        if remote_stat.st_size == local_stat.st_size:
410                            logger.info(f"文件已存在，跳过: {rom_file.name}")
411                            uploaded_count += 1
412                            continue
413                    except FileNotFoundError:
414                        pass
415                    
416                    # 上传文件
417                    logger.info(f"上传: {rom_file.name}")
418                    sftp.put(str(rom_file), remote_file)
419                    uploaded_count += 1
420                    
421                except Exception as e:
422                    logger.error(f"上传文件失败 {rom_file.name}: {e}")
423            
424            sftp.close()
425            ssh.close()
426            
427            logger.info(f"上传完成: {uploaded_count}/{len(rom_files)} 个文件")
428            return uploaded_count > 0
429            
430        except Exception as e:
431            logger.error(f"上传过程出错: {e}")
432            ssh.close()
433            return False

435    def run(self, search_query: str = "nes 100 in 1"):
436        """运行完整的下载和传输流程"""
437        logger.info("=== NES ROM 下载和传输工具 ===")
438
439        # 1. 优先尝试配置文件中的下载地址
440        nes_zip_urls = self.config.get("nes_zip_urls", [])
441        zip_path = None
442        download_url = None
443        if nes_zip_urls:
444            logger.info(f"配置文件中找到 {len(nes_zip_urls)} 个NES ZIP下载地址，优先尝试...")
445            for url in nes_zip_urls:
446                filename = url.split('/')[-1]
447                logger.info(f"尝试下载: {url}")
448                zip_path = self.download_file(url, filename)
449                if zip_path:
450                    download_url = url
451                    logger.info(f"成功下载: {url}")
452                    break
453                else:
454                    logger.warning(f"下载失败: {url}")
455        
456        # 2. 如果配置地址全部失败，则自动搜索
457        if not zip_path:
458            logger.info("配置地址全部失败，自动搜索可用ROM...")
459            results = self.search_roms(search_query)
460            if not results:
461                logger.error("未找到ROM文件")
462                return False
463            print("\n搜索结果:")
464            for i, result in enumerate(results[:5]):
465                print(f"{i+1}. {result['title']} (下载次数: {result['downloads']})")
466            selected_result = results[0]
467            logger.info(f"选择: {selected_result['title']}")
468            download_url = self.get_download_url(selected_result['identifier'])
469            if not download_url:
470                logger.error("无法获取下载链接")
471                return False
472            filename = download_url.split('/')[-1]
473            zip_path = self.download_file(download_url, filename)
474            if not zip_path:
475                logger.error("下载失败")
476                return False
477
478        # 3. 验证文件
479        if not self.verify_file(zip_path):
480            logger.error("文件验证失败")
481            return False
482
483        # 4. 解压ROM文件
484        rom_files = self.extract_roms(zip_path)
485        if not rom_files:
486            logger.error("解压失败或未找到ROM文件")
487            return False
488
489        # 5. 上传到树莓派
490        if self.upload_roms(rom_files):
491            logger.info("ROM传输完成！")
492            return True
493        else:
494            logger.error("ROM传输失败")
495            return False

213    def download_file(self, url: str, filename: str) -> Optional[Path]:
214        """下载文件，支持断点续传"""
215        file_path = self.download_dir / filename
216        
217        # 检查是否已存在
218        if file_path.exists():
219            logger.info(f"文件已存在: {file_path}")
220            return file_path
221        
222        try:
223            # 获取文件大小
224            response = self.session.head(url, timeout=30)
225            response.raise_for_status()
226            
227            total_size = int(response.headers.get('content-length', 0))
228            
229            # 检查是否有部分下载的文件
230            temp_path = file_path.with_suffix('.tmp')
231            resume_pos = 0
232            
233            if temp_path.exists():
234                resume_pos = temp_path.stat().st_size
235                if resume_pos >= total_size:
236                    # 文件已完整下载
237                    temp_path.rename(file_path)
238                    logger.info(f"文件下载完成: {file_path}")
239                    return file_path
240            
241            # 设置断点续传头
242            headers = {}
243            if resume_pos > 0:
244                headers['Range'] = f'bytes={resume_pos}-'
245                logger.info(f"断点续传: {resume_pos} bytes")
246            
247            # 开始下载
248            response = self.session.get(url, headers=headers, stream=True, timeout=30)
249            response.raise_for_status()
250            
251            mode = 'ab' if resume_pos > 0 else 'wb'
252            
253            with open(temp_path, mode) as f:
254                with tqdm(
255                    total=total_size,
256                    initial=resume_pos,
257                    unit='B',
258                    unit_scale=True,
259                    desc=filename
260                ) as pbar:
261                    
262                    for chunk in response.iter_content(chunk_size=self.config["download"]["chunk_size"]):
263                        if chunk:
264                            f.write(chunk)
265                            pbar.update(len(chunk))
266            
267            # 重命名临时文件
268            temp_path.rename(file_path)
269            logger.info(f"下载完成: {file_path}")
270            return file_path
271            
272        except Exception as e:
273            logger.error(f"下载失败: {e}")
274            # 清理临时文件
275            if temp_path.exists():
276                temp_path.unlink()
277            return None

377    def upload_roms(self, rom_files: List[Path]) -> bool:
378        """上传ROM文件到树莓派"""
379        if not rom_files:
380            logger.warning("没有ROM文件需要上传")
381            return False
382        
383        ssh = self.connect_sftp()
384        if not ssh:
385            return False
386        
387        try:
388            sftp = ssh.open_sftp()
389            pi_config = self.config["raspberry_pi"]
390            remote_path = pi_config["roms_path"]
391            
392            # 确保远程目录存在
393            try:
394                sftp.stat(remote_path)
395            except FileNotFoundError:
396                logger.info(f"创建远程目录: {remote_path}")
397                sftp.mkdir(remote_path)
398            
399            uploaded_count = 0
400            
401            for rom_file in rom_files:
402                try:
403                    remote_file = f"{remote_path}/{rom_file.name}"
404                    
405                    # 检查文件是否已存在
406                    try:
407                        remote_stat = sftp.stat(remote_file)
408                        local_stat = rom_file.stat()
409                        
410                        if remote_stat.st_size == local_stat.st_size:
411                            logger.info(f"文件已存在，跳过: {rom_file.name}")
412                            uploaded_count += 1
413                            continue
414                    except FileNotFoundError:
415                        pass
416                    
417                    # 上传文件
418                    logger.info(f"上传: {rom_file.name}")
419                    sftp.put(str(rom_file), remote_file)
420                    uploaded_count += 1
421                    
422                except Exception as e:
423                    logger.error(f"上传文件失败 {rom_file.name}: {e}")
424            
425            sftp.close()
426            ssh.close()
427            
428            logger.info(f"上传完成: {uploaded_count}/{len(rom_files)} 个文件")
429            return uploaded_count > 0
430            
431        except Exception as e:
432            logger.error(f"上传过程出错: {e}")
433            ssh.close()
434            return False
